196  template<typename Particles>
197  auto calc_potential_energy(Particles const &particles) {
198 
199  typename Particles::Scalar potential_eng{0};
200  size_t const size = particles.number();
201  auto const &m = particles.mass();
202  auto const &v = particles.vel();
203  auto const &p = particles.pos();
204 
205  if constexpr (HAS_METHOD(Particles, chain_pos) && HAS_METHOD(Particles, index)) {
206  auto const &ch_px = particles.chain_pos().x;
207  auto const &ch_py = particles.chain_pos().y;
208  auto const &ch_pz = particles.chain_pos().z;
209  auto const &idx = particles.index();
210 
211  for (size_t i = 0; i < size - 1; ++i) {
212  potential_eng -=
213  m[idx[i]] * m[idx[i + 1]] / sqrt(ch_px[i] * ch_px[i] + ch_py[i] * ch_py[i] + ch_pz[i] * ch_pz[i]);
214  }
215 
216  for (size_t i = 0; i < size - 2; ++i) {
217  auto dx = ch_px[i] + ch_px[i + 1];
218  auto dy = ch_py[i] + ch_py[i + 1];
219  auto dz = ch_pz[i] + ch_pz[i + 1];
220  potential_eng -= m[idx[i]] * m[idx[i + 2]] / sqrt(dx * dx + dy * dy + dz * dz);
221  }
222 
223  for (size_t i = 0; i < size; ++i) {
224  for (size_t j = i + 3; j < size; ++j) {
225  auto dx = p.x[idx[j]] - p.x[idx[i]];
226  auto dy = p.y[idx[j]] - p.y[idx[i]];
227  auto dz = p.z[idx[j]] - p.z[idx[i]];
228  potential_eng -= m[idx[i]] * m[idx[j]] / sqrt(dx * dx + dy * dy + dz * dz);
229  }
230  }
231  } else {
232  for (size_t i = 0; i < size; ++i)
233  for (size_t j = i + 1; j < size; ++j) {
234  auto dx = p.x[i] - p.x[j];
235  auto dy = p.y[i] - p.y[j];
236  auto dz = p.z[i] - p.z[j];
237  potential_eng -= m[i] * m[j] / sqrt(dx * dx + dy * dy + dz * dz);
238  }
239  }
240  return potential_eng;
241  }

254  template<typename ScalarArray, typename Coord>
255  inline auto calc_fall_free_time(ScalarArray const &mass, Coord const &position) {
256  using Scalar = typename ScalarArray::value_type;
257  size_t const size = mass.size();
258  Scalar min_fall_free = max_value<Scalar>::value;
259 
260  for (size_t i = 0; i < size; i++) {
261  for (size_t j = i + 1; j < size; j++) {
262  Scalar dx = position.x[i] - position.x[j];
263  Scalar dy = position.y[i] - position.y[j];
264  Scalar dz = position.z[i] - position.z[j];
265  Scalar r = sqrt(dx * dx + dy * dy + dz * dz);
266  Scalar fall_free = pow(r, 1.5) / sqrt(mass[i] + mass[j]);
267 
268  if (fall_free < min_fall_free) min_fall_free = fall_free;
269  }
270  }
271  return min_fall_free * space::consts::pi * 0.5 / sqrt(2 * space::consts::G);
272  }
