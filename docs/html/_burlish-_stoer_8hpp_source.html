41  explicit BurlishStoerConsts() {
42  for (size_t i = 0; i < MaxIter; ++i) {
43  sub_steps_[i] = 2 * (i + 1);
44 
45  if (i == 0) {
46  cost_[i] = sub_steps_[i] + 1;//The additional 1 is for 'KDK' method.
47  } else {
48  cost_[i] = cost_[i - 1] + sub_steps_[i] + 1;
49  }
50 
51  for (size_t j = 0; j < i; ++j) {
52  Scalar ratio = static_cast<Scalar>(sub_steps_[i]) / static_cast<Scalar>(sub_steps_[j]);
53  extrap_coef_[at(i, j)] = 1.0 / (ratio * ratio - 1.0);
54  }
55  }
56  }

94  class BurlishStoer : public OdeIterator<BurlishStoer<Real, ErrChecker, StepControl>> {
95  static_assert(std::is_floating_point<Real>::value, "Only float-like type can be used!");
96  public:

111  template<typename U>
112  auto impl_iterate(U &particles, typename U::Scalar macro_step_size) -> typename U::Scalar {
113  static_assert(is_particle_system_v<U>, "Passing non particle-system-type!");
114 
115  Scalar iter_h = macro_step_size;
116  particles.to_linear_container(input_);
117  check_variable_size();
118 
119  for (size_t i = 0; i < max_try_num; ++i) {
120  iter_num_++;
121  bool reject = true;
122 
123  integrate_by_n_steps(particles, iter_h, parameters_.step_sequence(0));
124  particles.to_linear_container(extrap_list_[0]);
125 
126  for (size_t k = 1; k <= ideal_rank_ + 1; ++k) {
127  particles.load_from_linear_container(input_);
128  integrate_by_n_steps(particles, iter_h, parameters_.step_sequence(k));
129  particles.to_linear_container(extrap_list_[k]);
130  extrapolate(k);
131 
132  Scalar error = err_checker_.error(input_, extrap_list_[0], extrap_list_[1]);
133 
134  ideal_step_size_[k] = step_controller_.next_step_size(2 * k + 1, iter_h, std::make_tuple(error, last_error_));
135 
136  cost_per_len_[k] = parameters_.cost(k) / ideal_step_size_[k];
137  //space::print_csv(std::cout, k, ideal_rank_, error, ideal_step_size_[k], cost_per_len_[k],'\n');
138  if (in_converged_window(k)) {
139  if (error < 1.0) {
140  reject = false;
141  iter_h = set_next_iteration(k, last_step_reject_);
142  particles.load_from_linear_container(extrap_list_[0]);
143  last_step_reject_ = reject;
144  last_error_ = error;
145  return iter_h;
146  } else if (is_diverged_anyhow(error, k)) {
147  reject = true;
148  rej_num_++;
149  iter_h = get_next_try_step(k);
150  break;
151  }
152  }
153  }
154  last_step_reject_ = reject;
155  particles.load_from_linear_container(input_);
156  }
157 
158  spacehub_abort("Reach max iteration loop number!");
159  }

184  template<typename U>
185  void integrate_by_n_steps(U &particles, Scalar macro_step_size, size_t steps) {
186  Scalar h = macro_step_size / steps;
187  particles.kick(0.5 * h);
188  for (size_t i = 1; i < steps; i++) {
189  particles.drift(h);
190  particles.kick(h);
191  }
192  particles.drift(h);
193  particles.kick(0.5 * h);
194  }

196  void extrapolate(size_t k) {
197  for (size_t j = k; j > 0; --j) {
198  auto c_1 = 1 + parameters_.table_coef(k, j - 1);
199  auto c_2 = -parameters_.table_coef(k, j - 1);
200  for (size_t i = 0; i < var_num_; ++i) {
201  extrap_list_[j - 1][i] = c_1 * extrap_list_[j][i] + c_2 * extrap_list_[j - 1][i];
202  }
203  }

224  Scalar set_next_iteration(size_t k, bool last_reject) {
225  if (k == ideal_rank_) {
226  if (cost_per_len_[k - 1] < BSConsts::cost_tol * cost_per_len_[k]) {
227  ideal_rank_ = allowed(k - 1);
228  return ideal_step_size_[ideal_rank_];
229  } else if (cost_per_len_[k] < BSConsts::cost_tol * cost_per_len_[k - 1] && !last_reject) {
230  ideal_rank_ = allowed(k + 1);
231  return ideal_step_size_[k] * static_cast<Scalar>(parameters_.cost(ideal_rank_)) /
232  static_cast<Scalar>(parameters_.cost(k));
233  } else {
234  return ideal_step_size_[ideal_rank_];
235  }
236  } else if (k == ideal_rank_ - 1) {
237  if (ideal_rank_ <= 2 || cost_per_len_[k] < BSConsts::cost_tol * cost_per_len_[k - 1]) {
238  ideal_rank_ = allowed(k + 1);
239  return ideal_step_size_[k] * static_cast<Scalar>(parameters_.cost(k + 1)) /
240  static_cast<Scalar>(parameters_.cost(k));
241  } else {
242  ideal_rank_ = allowed(k);
243  return ideal_step_size_[k];
244  }
245  } else if (k == ideal_rank_ + 1) {
246  if (cost_per_len_[k - 2] < BSConsts::cost_tol * cost_per_len_[k - 1]) {
247  ideal_rank_ = allowed(ideal_rank_ - 1);
248  }
249  if (cost_per_len_[k] < BSConsts::cost_tol * cost_per_len_[ideal_rank_] && !last_reject) {
250  ideal_rank_ = allowed(k);
251  }
252  return ideal_step_size_[ideal_rank_];
253  } else {
254  spacehub_abort("unexpected iteration index!");
255  }
256  }

258  bool is_diverged_anyhow(Scalar error, size_t k) const {
259  Scalar r = 1.0;
260 
261  if (k == ideal_rank_ - 1) {
262  r = static_cast<Scalar>(parameters_.step_sequence(k + 1) * parameters_.step_sequence(k + 2)) /
263  static_cast<Scalar>(parameters_.step_sequence(0) * parameters_.step_sequence(0));
264  } else if (k == ideal_rank_) {
265  r = static_cast<Scalar>(parameters_.step_sequence(k)) / static_cast<Scalar>(parameters_.step_sequence(0));
266  } // else k == iterDepth+1 and error >1 reject directly
267 
268  return error > r * r;
269  }
